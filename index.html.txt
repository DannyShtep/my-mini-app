  <script>
    const canvas = document.getElementById("wheel");
    const ctx = canvas.getContext("2d");
    let angle = 0;
    let spinning = false;

    function drawWheel() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      const colors = ['#ff4444', '#44ff44', '#4444ff', '#ffaa00', '#00ffaa', '#ff00aa'];
      const segments = 6;
      const arc = (2 * Math.PI) / segments;

      for (let i = 0; i < segments; i++) {
        const start = angle + i * arc;
        ctx.beginPath();
        ctx.moveTo(canvas.width / 2, canvas.height / 2);
        ctx.arc(canvas.width / 2, canvas.height / 2, canvas.width / 2 - 10, start, start + arc);
        ctx.fillStyle = colors[i % colors.length];
        ctx.fill();
        ctx.closePath();
      }
    }

    function easeOut(t) {
      return 1 - Math.pow(1 - t, 4);
    }

    let playersJoined = 0;
    const requiredPlayers = 2;
    let gameStarted = false;
    let countdownTimer = null;

    function startCountdown() {
      let duration = 15;
      const fill = document.getElementById("progressFill");
      const label = document.getElementById("timerCount");

      clearInterval(countdownTimer);
      countdownTimer = setInterval(() => {
        duration--;
        label.textContent = `${duration}s`;
        fill.style.width = `${(100 * (15 - duration)) / 15}%`;
        if (duration <= 0) {
          clearInterval(countdownTimer);
          spinWheel();
        }
      }, 1000);
    }

    function simulatePlayerJoin() {
      playersJoined++;
      if (playersJoined >= requiredPlayers && !gameStarted) {
        gameStarted = true;
        startCountdown();
      }
    }

    simulatePlayerJoin(); // simulating player 1
    setTimeout(simulatePlayerJoin, 3000); // simulating player 2

    function spinWheel() {
      spinning = true;
      const duration = 4000;
      const start = performance.now();
      const fullRotations = 4;
      const targetAngle = Math.random() * 2 * Math.PI;

      function animate(now) {
        const elapsed = now - start;
        const progress = Math.min(elapsed / duration, 1);
        const eased = easeOut(progress);
        angle = (2 * Math.PI * fullRotations * eased + targetAngle) % (2 * Math.PI);
        drawWheel();

        if (progress < 1) {
          requestAnimationFrame(animate);
        } else {
          spinning = false;
          showWinnerModal({
            username: "grinch",
            photo: "https://api.dicebear.com/7.x/identicon/svg?seed=grinch",
            totalTon: 42,
            nftImages: [
              "https://via.placeholder.com/40",
              "https://via.placeholder.com/40"
            ]
          });
          setTimeout(() => {
            gameStarted = false;
            playersJoined = 0;
            document.getElementById("progressFill").style.width = "0%";
            document.getElementById("timerCount").textContent = "0s";
            simulatePlayerJoin();
            setTimeout(simulatePlayerJoin, 2000);
          }, 4200);
        }
      }

      requestAnimationFrame(animate);
    }

    drawWheel();

    const connectBtn = document.getElementById('tonConnectBtn');
    const balanceBox = document.getElementById('tonBalance');
    const connector = new TON_CONNECT.TonConnect();

    connectBtn.onclick = async () => {
      await connector.connectWallet();
    };

    connector.onStatusChange(async wallet => {
      if (wallet?.account?.address) {
        connectBtn.style.display = "none";
        balanceBox.style.display = "block";

        const address = wallet.account.address;
        const res = await fetch(`https://toncenter.com/api/v2/getAddressBalance?address=${address}`);
        const data = await res.json();
        const tonBalance = Number(data.result) / 1e9;
        balanceBox.textContent = `Баланс: ${tonBalance.toFixed(2)} TON`;
      }
    });

    function showWinnerModal({ username, photo, totalTon, nftImages }) {
      document.getElementById('winnerPhoto').src = photo;
      document.getElementById('winnerName').textContent = '@' + username;
      document.getElementById('winnerAmount').textContent = `Ставка: ${totalTon} TON`;
      const icons = document.getElementById('nftIcons');
      icons.innerHTML = '';
      nftImages.forEach(img => {
        const icon = document.createElement('img');
        icon.src = img;
        icons.appendChild(icon);
      });
      document.getElementById('resultModal').style.display = 'block';
      setTimeout(() => {
        document.getElementById('resultModal').style.display = 'none';
      }, 4000);
    }

    document.getElementById('closeModal').onclick = () => {
      document.getElementById('resultModal').style.display = 'none';
    };
